\prototype{\dfsProc{\Tree \(t\)}}{
	\If{\(t \neq \Nil\)}{

		\BlankLine
		\tcp{pre-order visit}
		\Print \(t.\Value\)\;

		\BlankLine
		\dfsProc{\(t.\treeLeft\)}\;

		\BlankLine
		\tcp{in-order visit}
		\Print \(t.\Value\)\;

		\BlankLine
		\dfsProc{\(t.\treeRight\)}\;

		\BlankLine
		\tcp{post-order visit}
		\Print \(t.\Value\)\;
	}
}

\end{minipage}%
