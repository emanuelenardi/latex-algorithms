% NOTE simbolo di ricorrenza
\NewDocumentCommand{\T}{ s o m }{
	\ensuremath{%
		\IfValueTF{#2}
			{T_{#2}}
			{T}
		\left(%
		\IfBooleanTF{#1}
			{\left\lfloor#3\right\rfloor}
			{#3}
		\right)
	}
}
