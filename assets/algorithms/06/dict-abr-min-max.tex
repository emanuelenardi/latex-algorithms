\BlankLine
\tcp{RICERCA DEL MINIMO}
\prototype{\Tree \minNode{\Tree \(T\)}}{
	\Tree \(u = T\) \Comment*[l]{parto dalla radice}

	\BlankLine
	\While{\(u.\varLeft \Neq \Nil\)}{
		\(u \Assign u.\varLeft\)\;
	}

	\BlankLine
	\Return \(u\)\;
}

\vspace{-5pt}
\vphantom{0pt}

\BlankLine
\tcp{RICERCA DEL MASSIMO}
\prototype{\Tree \maxNode{\Tree \(T\)}}{
	\Tree \(u = T\) \Comment*[l]{parto dalla radice}

	\BlankLine
	\While{\(u.\varRight \Neq \Nil\)}{
		\(u \Assign u.\varRight\)\;
	}

	\BlankLine
	\Return \(u\)\;
}
