\BlankLine
\tcp{determina la distanza \(d(x,y) = \abs{x-y}\) minima fra due elementi di un albero binario di ricerca}
\fi

\prototype{\Int \minDist{\Tree t}}{
	\Tree \(u = t.\minFunction\)\;
	\Int \(\varMin = +\infty\)\;
	\Int \(\varPrev = -\infty\)\;

	\BlankLine
	\While{\(u \Neq \Nil\)}{
		\If{\(u.\varValue - \varPrev < \varMin\)}{
			\(\varMin = u.\varValue - \varPrev\)\;
		}
		\(\varPrev = u.\varValue\)\;
		\(u = u.\succNode\)\;
	}
}
