\prototype{\Int \stampaEspressioni{\Tree \(t\)}}{
	\eIf{\(t.\treeLeft \Equal \Nil\) \And \(t.\treeRight \Equal \Nil\)}{
		\tcp{siamo in una foglia}
		\Print \(t.\treeRead\)
	}{
		\tcp{sono su un nodo interno}
		\Print \enquote{(}\;

		\stampaEspressioni{\(t.\treeLeft\)}\;

		\Print \(t.\treeRead\)\;

		\stampaEspressioni{\(t.\treeRight\)}\;

		\Print \enquote{)}\;
	}
}
