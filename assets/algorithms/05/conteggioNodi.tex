\prototype{\conteggioNodi{\Tree \(t\)}}{
	\eIf{\(t \Equal \Nil\)}{
		\tcp{Ã¨ un albero vuoto}
		\Return \(0\)\;
	}{
		\tcp{conto ricorsivamente i nodi}
		\(C_{\ell} = \conteggioNodi{\(t.\treeLeft\)}\)\;
		\(C_{r} = \conteggioNodi{\(t.\treeRight\)}\)\;

		\BlankLine
		\Return \(C_{\ell} + C_{r} + 1\)\;
	}
}
