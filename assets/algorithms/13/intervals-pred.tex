\tcp{Pre-computa i predecessori}
\prototype{\Array{\Int} \computePredecessor{\Array{\Int} a, \Array{\Int} a, \Int n}}{

	\BlankLine
	\Matrix{\Int} \(pred\) \(\Assign\) \new \Array{\Int}{0}{n}\;
	\(pred[0] \Assign 0\)\;

	\BlankLine
	\From{\(i \Assign 1\) \DownTo \(n\)}{
		\(j \Assign i-1\)\;

		\BlankLine
		\While{\(j > 0\) \And \(b[j] > a[i]\)}{
			\(j \Assign j - 1\)\;
		}
		\(pred[i] \Assign j\)\;
	}

	\BlankLine
	\Return \(pred\)\;
}

\end{algorithm}
