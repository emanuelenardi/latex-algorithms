\SetKwFunction{hateville}{hateville}

\prototype{\Int \hateville{\Array{\Int} D, \Int n}}{

	\BlankLine
	\Array{\Int} \(DP\) \Assign \new \Array{\Int}{0}{n}\;
	\(DP[0] \Assign 0\)\;
	\(DP[1] \Assign DP[1]\)\;

	\BlankLine
	\From{\(i \Assign 2\) \DownTo \(n\)}{
		\(DP[i]\) \Assign \maxFunction{\(DP[i-1]\), \(DP[i-2] + D[i]\)}
	}

	\BlankLine
	\Return \(DP[n]\)\;
}
