\caption[Fibonacci iterativo]{}

\prototype{\Int \fibonacciIter{\Int \(n\)}}{

	\BlankLine
	\(DP\) \Assign \new \Array{\Int}{0}{n}\;
	\(DP[0] \Assign DP[1] \Assign 1\)\;

	\BlankLine
	\From{\(i \Assign 2\) \DownTo \(n\)}{
		\(DP[i]\) \Assign \(DP[i-1] + DP[i-2]\)\;
	}

	\BlankLine
	\Return \(DP[n]\)\;
}
