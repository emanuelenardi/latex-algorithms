\caption[Moltiplicazione fra matrici]{}
\fi

\prototype{\Matrix{\Int} \multiplyFunction{\Matrix{\Int} A, \Matrix{\Int} S, \Int i, \Int j}}{

	\BlankLine
	\eIf{\( i \Equal j \)}
	{ \Return \Array{A}{i}\; }
	{
		\BlankLine
		\Matrix{\Int} \(X\) = \multiplyFunction{\( \MatrixCall{last}{}{}, i, \MatrixCall{last}{i}{j} \)}\;

		\Matrix{\Int} \(Y\) = \multiplyFunction{\( \MatrixCall{last}{}{}, \MatrixCall{last}{i}{j} + 1, j \)}\;

		\BlankLine
		\Return \multiplyMatrices{X,Y}
	}
}

\end{algorithm}
