\prototype{\donGiovanni{\Sequence catalogo}}{
	\Sequence \(L \Assign \sequenceConstructor\)\;
	\Postype \(q \Assign catalogo.\listHead\)\;

	\BlankLine
	\While{\Not \(catalogo.\listEnd\)}{
		\Item \(v \Assign catalogo.\listRead{q}\)\;

		\BlankLine
		\lIf{\(v.nazione \Equal spagnola\)}{\(L.\listInsert{L.\listTail.\listSucc, v}\)}

		\BlankLine
		\eSea{\(v.rango \Equal duchessa\)}{
		 	\(q \Assign catalogo.\listRemove{q}\)\;
		}{
			\(q \Assign catalogo.\listSucc{q}\)\;
		}
	}
}

\end{algorithm}
