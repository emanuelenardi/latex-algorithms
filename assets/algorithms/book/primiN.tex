\begin{algorithm}[H]
\caption{Costruisce una lista con all'interno i primi \(n\) numeri}
\fi

\prototype{\primiN{\Int n, \List L, \List M}}{

	\BlankLine
	\If{\( n \geqslant 1 \)}{
		\(L.\listInsert{L.\listHead{}, n}\)\;
		\(M.\listInsert{M.\listTail{}.\listSucc{}, n}\)\;

		\BlankLine
		\tcp{chiamata ricorsiva}
		\primiN{\(n-1\), L, M}\;
	}
}

\end{algorithm}
