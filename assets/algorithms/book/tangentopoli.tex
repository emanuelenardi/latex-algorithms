\prototype{\tangentopoli{\List L}}{
	\Postype \(p \Assign L.\listHead\)\;

	\BlankLine
	\While{\Not \(L.\listEmpty\)}{
		\From{\(i \Assign 1\) \DownTo \(k-1\)}{
			\(p \Assign L.\listSucc{p}\)\;

			\BlankLine
			\lIf{\(L.\listEnd{p}\)}{
				\(p \Assign L.\listHead\)
			}

			\BlankLine
			\Print \(L.\listRead{p}\)\;
			\(p \Assign L.\listRemove{p}\)\;
		}
	}
}
