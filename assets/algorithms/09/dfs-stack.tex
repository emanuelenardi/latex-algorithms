\BlankLine
\tcp{Algoritmo iteratico, \Stack esplicito, pre-order \slide{47}{101}}
\prototype{\dfsStack{\Graph \(G\), \Node \(r\)}}{
	\Stack \(S\) \Assign \stackConstructor\;
	\(S\).\stackPush\;

	\BlankLine
	\tcp{inizializzazione}
	\Array{\Bool} \(visitato\) \Assign \Array{\Bool}{1}{G.n}\;
	\lForEach{\(u \in G.\VV - \{r\}\)}{
		\ArrayCall{visitato}{u} \Assign \False\;
	}
	\ArrayCall{visitato}{r} \Assign \True \Comment*[l]{radice visitata}

	\While{\Not \(S\).\setEmpty}{
		\Node \(u\) \Assign \(S\).\stackPop\;

		\BlankLine
		\If{\Not \ArrayCall{visitato}{v}}{

			\BlankLine
			\{ esamina il nodo \(u\) (\emph{pre}ordine) \}\;
			\dots
		}
	}
}

\end{algorithm}
