\tcp{genera un albero breadth-first}
\prototype{\dfsProc{\Graph G, \Node u, \Array{\Bool} visitato}}{
	\(\ArrayCall{visitato}{u} = \True\)\;

	\BlankLine
	\lnl{dfs:pre-visita}%
	\{ esamina il nodo \(u\) (caso \textbf{pre}visita) \}

	\BlankLine
	\ForEach{\(u \in G.\adj{u}\)}{

		\BlankLine
		\{ esamina l'arco \((u,v)\) \}

		\BlankLine
		\If{\Not \ArrayCall{visitato}{u}}{
			\tcp{chiamata ricorsiva}
			\depthFirstSearch{G, v, visitato}
		}
	}

	\BlankLine
	\lnl{dfs:post-visita}%
	\{ esamina il nodo \(u\) (caso \textbf{post}visita)\}
}

\ifstandalone
\end{algorithm}
