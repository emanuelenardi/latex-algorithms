\begin{algorithm}[H]
\caption[Visita in profondità iterativa]{}

La procedura si ottiene semplicemente sostituendo una pila alla coda utilizzata nella visita \breadthFirstSearch.
\fi


\BlankLine
\tcp{effettua una visita in profondità iterativa}
\prototype{\dfsProcIter{\Graph G, \Node r}}{
	\Stack \(S = \stackConstructor\)\;
	\(S.\stackPush{r}\)\;

	\BlankLine
	\Array{\Bool} \(visitato =\) \new \Array{\Bool}{1}{G.\graphSize}\;
	\lForEach{\(u \in G.\VV - \{r\}\)}{
		\(visitato[u] = \False\)
	}

	\BlankLine
	\(visitato[r] = \True\)\;
	\While{\Not \(S.\stackEmpty\)}{
		\Node \(u = S.\stackPop\)\;

		\BlankLine
		\{ esamina il nodo \(u\) \}

		\BlankLine
		\ForEach{\(v \in G.\adj{u}\)}{

			\BlankLine
			\{ esamina l'arco \((u,v)\) \}

			\BlankLine
			\If{\Not \(visitato[v]\)}{
				\(visitato[v] = \True\)\;
				\(S.\stackPush{v}\)\;
			}
		}
	}
}

\ifstandalone
\end{algorithm}
