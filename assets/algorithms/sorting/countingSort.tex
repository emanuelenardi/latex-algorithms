\prototype{\countingSort{\Item{} A, \Int n, \Int k}}{
	\Int i, j, k\;
	\Array{\Int} \(B\) \Assign \new \Array{\Int}{1}{k}\;

	\BlankLine
	\tcp{Inizializza i vettori}
	\lFor{\(i \Assign 1\) \DownTo \(k\)}{\(B[i] = 0\)}
	\lFor{\(j \Assign 1\) \DownTo \(n\)}{\(B[A[j]] = B[A[j]] + 1\)}

	\BlankLine
	\(j = 1\)\;
	\From{\(i=1\) \DownTo \(k\)}{
		\While{\(B[i] > 0\)}{
			\(A[j] \Assign i\)\;
			\(\Increment{j}\)\;
			\(B[i] \Assign B[i] - 1\)\;
		}
	}
}
