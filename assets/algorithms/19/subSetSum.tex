\prototype{\Bool \subSetSum{\Array{\Int} \(A\), \Int \(n\), \Int \(k\)}}{

	\BlankLine
	\Array{\Bool} \(DP\) \Assign \new \Array{\Bool}{0}{k}\;

	\BlankLine
	\tcp{casi base}

	\tcp{dati nessun oggetto posso ottenere il valore 0}
	\(DP[0][0] \Assign \True\)\;

	\For{\(r \Assign 1\) \DownTo \(k\)}{
		\tcp{dati 0 oggetti posso ottenere un valore di \(k > 0\)}
		\(DP[0][r] \Assign \False\)\;
	}

	\BlankLine
	\tcp{riempi la tabella}
	\From{\(i \Assign 1\) \DownTo \(n\)}{
		\From(\Comment*[h]{\(A[i]\) perch√©}){\(r \Assign A[i]\) \DownTo \(k\)}{
		\tcp{\(A[i]\): semplifico}

			\BlankLine
			\tcp{logica dell'algoritmo}
			\(DP[i][r] \Assign DP[i-1][k]\) \Or \(DP[i-1][r - A[i]]\)\;
		}
	}

	\BlankLine
	\Return DP[n][k]\;
}
